async function inviaTrasmissione() { const keyInput = document.getElementById('api-key').value.trim(); const textInput = document.getElementById('prompt').value.trim(); const resDiv = document.getElementById('response'); if (!keyInput || !textInput) { resDiv.innerHTML = "Errore: Incolla l'API Key e inserisci un messaggio!"; return; } resDiv.classList.remove('error'); resDiv.innerHTML = "Cortana: Elaborazione in corso..."; const promptFuso = `Tu sei Cortana di Halo, l'IA tattica di supporto dell'utente. Sei militare, leale e ironica. Rispondi in italiano in modo breve (massimo due frasi): ${textInput}`; try { // Utilizziamo il modello stabile standard 1.5-flash per evitare i 503 di sovraccarico const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${keyInput}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: promptFuso }] }] }) }); if (!res.ok) throw new Error(`Errore Server Google: ${res.status}`); const data = await res.json(); const testoRisposta = data.candidates[0].content.parts[0].text; resDiv.innerHTML = "Cortana: " + testoRisposta; document.getElementById('prompt').value = ""; // RIPRODUZIONE VOCALE NATIVA FEMMINILE if ('speechSynthesis' in window) { window.speechSynthesis.cancel(); const utterance = new SpeechSynthesisUtterance(testoRisposta); utterance.lang = 'it-IT'; utterance.rate = 1.05; utterance.pitch = 1.25; // Alziamo il pitch per forzare una tonalità femminile anche su voci neutre const voices = window.speechSynthesis.getVoices(); // Filtro stringente per acchiappare Elsa, Alice, Google o Microsoft in italiano const voceFemminile = voices.find(v => v.lang.includes('it') && (v.name.toLowerCase().includes('elsa') || v.name.toLowerCase().includes('alice') || v.name.toLowerCase().includes('federica') || v.name.toLowerCase().includes('google'))) || voices.find(v => v.lang.includes('it')); if (voceFemminile) utterance.voice = voiceFemminile; window.speechSynthesis.speak(utterance); resDiv.innerHTML += "
[Audio riprodotto via sistema locale]"; } } catch (error) { console.error(error); resDiv.classList.add('error'); resDiv.innerHTML = `Errore di trasmissione: ${error.message}`; } }
